       expect(instance).toBeInstanceOf(ConsoleLogger);
     });

         constructor(

           public database: Database
         ) {}
       }

       const adapter = createAdapter({
         provides: UserServicePort,
         requires: [LoggerPort, DatabasePort],
         lifetime: "scoped",
         class: UserServiceImpl,
       });

       const mockLogger = { log: vi.fn() };

     });
