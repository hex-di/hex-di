<task id="2" title="Create span adapter unit tests">
Test convertToReadableSpan field mapping accuracy and type conversions.

**File:** `packages/tracing-otel/tests/unit/span-adapter.test.ts`

**Tests to write:**

1. **Field mapping:** All HexDI fields map to correct OTel fields
2. **HrTime conversion:** Milliseconds convert to [seconds, nanoseconds]
3. **SpanKind mapping:** internal/server/client enum conversion
4. **SpanStatus mapping:** unset/ok/error conversion
5. **SpanEvent conversion:** Events with time, name, attributes
6. **SpanLink conversion:** Links with traceId, spanId, attributes
7. **ParentSpanContext:** Generated from parentSpanId when present
8. **TraceState handling:** Immutable TraceState wrapper
9. **Default resource:** Creates "hex-di-app" when not provided
10. **Custom resource:** Uses provided resource when given

**Test structure:**

```typescript
import { describe, it, expect } from "vitest";
import { convertToReadableSpan } from "@hex-di/tracing-otel";
import { SpanKind } from "@opentelemetry/api";
import type { SpanData } from "@hex-di/tracing";

describe("convertToReadableSpan", () => {
  const baseSpanData: SpanData = {
    context: {
      traceId: "trace123",
      spanId: "span456",
      traceFlags: 1,
      traceState: undefined,
    },
    name: "test-span",
    kind: "internal",
    startTime: 1000,
    endTime: 2000,
    status: "ok",
    attributes: { key: "value" },
    events: [],
    links: [],
    parentSpanId: undefined,
  };

  it("should map all basic fields correctly", () => {
    const otelSpan = convertToReadableSpan(baseSpanData);

    expect(otelSpan.name).toBe("test-span");
    expect(otelSpan.kind).toBe(SpanKind.INTERNAL);
    expect(otelSpan.spanContext().traceId).toBe("trace123");
    expect(otelSpan.spanContext().spanId).toBe("span456");
    expect(otelSpan.attributes).toEqual({ key: "value" });
    expect(otelSpan.ended).toBe(true);
  });

  it("should convert milliseconds to HrTime", () => {
    const otelSpan = convertToReadableSpan(baseSpanData);

    // 1000ms = [1, 0]
    expect(otelSpan.startTime).toEqual([1, 0]);
    // 2000ms = [2, 0]
    expect(otelSpan.endTime).toEqual([2, 0]);
    // duration = 1000ms = [1, 0]
    expect(otelSpan.duration).toEqual([1, 0]);
  });

  it("should convert span kinds correctly", () => {
    const serverSpan = { ...baseSpanData, kind: "server" as const };
    const clientSpan = { ...baseSpanData, kind: "client" as const };

    expect(convertToReadableSpan(serverSpan).kind).toBe(SpanKind.SERVER);
    expect(convertToReadableSpan(clientSpan).kind).toBe(SpanKind.CLIENT);
  });

  it("should generate parentSpanContext when parentSpanId present", () => {
    const childSpan = { ...baseSpanData, parentSpanId: "parent789" };
    const otelSpan = convertToReadableSpan(childSpan);

    expect(otelSpan.parentSpanContext).toBeDefined();
    expect(otelSpan.parentSpanContext?.spanId).toBe("parent789");
    expect(otelSpan.parentSpanContext?.traceId).toBe("trace123");
  });
});
```

<task id="3" title="Create batch processor unit tests">
Test buffering, batching, scheduled flush, and FIFO drop policy.

**File:** `packages/tracing-otel/tests/unit/batch-processor.test.ts`

**Tests to write:**

1. **Buffer accumulation:** Spans buffered up to maxQueueSize
2. **FIFO drop policy:** Oldest span dropped when buffer full
3. **Immediate flush:** Flush triggered when batch size reached
4. **Scheduled flush:** Flush after scheduledDelayMillis
5. **Multiple batches:** Large buffers exported in chunks
6. **Shutdown flush:** All buffered spans exported on shutdown
7. **Timeout protection:** Exporter shutdown with timeout
8. **Error handling:** Export errors logged but don't throw
9. **No-op after shutdown:** onEnd becomes no-op after shutdown

**Timer mocking setup:**

```typescript
import { vi, describe, it, expect, beforeEach, afterEach } from "vitest";
import { createBatchSpanProcessor } from "@hex-di/tracing-otel";
import type { SpanExporter, SpanData } from "@hex-di/tracing";

describe("createBatchSpanProcessor", () => {
  let mockExporter: SpanExporter;
  let exportedBatches: SpanData[][];

  beforeEach(() => {
    vi.useFakeTimers();
    exportedBatches = [];

    mockExporter = {
      export: vi.fn().mockImplementation((spans: SpanData[]) => {
        exportedBatches.push([...spans]);
        return Promise.resolve();
      }),
      forceFlush: vi.fn().mockResolvedValue(undefined),
      shutdown: vi.fn().mockResolvedValue(undefined),
    };
  });

  afterEach(() => {
    vi.useRealTimers();
  });

  it("should buffer spans and flush on schedule", async () => {
    const processor = createBatchSpanProcessor(mockExporter, {
      scheduledDelayMillis: 5000,
      maxExportBatchSize: 10,
    });

    // Add spans (below batch size)
    const span1 = createTestSpan("span1");
    const span2 = createTestSpan("span2");

    processor.onEnd(span1);
    processor.onEnd(span2);

    // No immediate export
    expect(mockExporter.export).not.toHaveBeenCalled();

    // Advance time to trigger scheduled flush
    await vi.advanceTimersByTimeAsync(5000);

    // Verify export called with buffered spans
    expect(mockExporter.export).toHaveBeenCalledTimes(1);
    expect(exportedBatches[0]).toHaveLength(2);
    expect(exportedBatches[0].map(s => s.name)).toEqual(["span1", "span2"]);

    await processor.shutdown();
  });

  it("should flush immediately when batch size reached", async () => {
    const processor = createBatchSpanProcessor(mockExporter, {
      maxExportBatchSize: 2,
      scheduledDelayMillis: 5000,
    });

    processor.onEnd(createTestSpan("span1"));
    processor.onEnd(createTestSpan("span2"));

    // Immediate flush (no timer needed)
    await vi.runAllTimersAsync();

    expect(mockExporter.export).toHaveBeenCalledTimes(1);
    expect(exportedBatches[0]).toHaveLength(2);

    await processor.shutdown();
  });

  it("should drop oldest span when buffer full (FIFO)", async () => {
    const processor = createBatchSpanProcessor(mockExporter, {
      maxQueueSize: 3,
      maxExportBatchSize: 10,
      scheduledDelayMillis: 5000,
    });

    // Fill buffer beyond capacity
    processor.onEnd(createTestSpan("span1"));
    processor.onEnd(createTestSpan("span2"));
    processor.onEnd(createTestSpan("span3"));
    processor.onEnd(createTestSpan("span4")); // This drops span1

    await processor.forceFlush();

    // Only spans 2, 3, 4 should be exported
    expect(exportedBatches[0].map(s => s.name)).toEqual(["span2", "span3", "span4"]);

    await processor.shutdown();
  });
});

function createTestSpan(name: string): SpanData {
  return {
    context: { traceId: "trace", spanId: name, traceFlags: 1 },
    name,
    kind: "internal",
    startTime: Date.now(),
    endTime: Date.now() + 100,
    status: "ok",
    attributes: {},
    events: [],
    links: [],
    parentSpanId: undefined,
  };
}
```

<task id="4" title="Create simple processor unit tests">
Test immediate export behavior and error handling.

**File:** `packages/tracing-otel/tests/unit/simple-processor.test.ts`

**Tests to write:**

1. **Immediate export:** Export called immediately on span end
2. **Fire-and-forget:** Export is non-blocking
3. **Shutdown delegation:** Calls exporter.shutdown with timeout
4. **Error handling:** Export errors logged but don't throw
5. **No-op after shutdown:** onEnd becomes no-op

**Test structure:**

```typescript
describe("createSimpleSpanProcessor", () => {
  let mockExporter: SpanExporter;
  let consoleErrorSpy: vi.SpyInstance;

  beforeEach(() => {
    mockExporter = {
      export: vi.fn().mockResolvedValue(undefined),
      forceFlush: vi.fn().mockResolvedValue(undefined),
      shutdown: vi.fn().mockResolvedValue(undefined),
    };

    consoleErrorSpy = vi.spyOn(console, "error").mockImplementation(() => {});
  });

  afterEach(() => {
    consoleErrorSpy.mockRestore();
  });

  it("should export immediately on span end", () => {
    const processor = createSimpleSpanProcessor(mockExporter);
    const span = createTestSpan("test");

    processor.onEnd(span);

    // Export called immediately (synchronously)
    expect(mockExporter.export).toHaveBeenCalledTimes(1);
    expect(mockExporter.export).toHaveBeenCalledWith([span]);
  });

  it("should not block on export (fire-and-forget)", async () => {
    // Make export slow
    let resolveExport: () => void;
    const exportPromise = new Promise<void>(resolve => {
      resolveExport = resolve;
    });

    mockExporter.export = vi.fn(() => exportPromise);

    const processor = createSimpleSpanProcessor(mockExporter);
    const span = createTestSpan("test");

    // This should return immediately
    const start = Date.now();
    processor.onEnd(span);
    const duration = Date.now() - start;

    expect(duration).toBeLessThan(10); // Should be ~0ms

    // Resolve export
    resolveExport!();
    await exportPromise;

    await processor.shutdown();
  });

  it("should handle export errors gracefully", () => {
    mockExporter.export = vi.fn().mockRejectedValue(new Error("Export failed"));

    const processor = createSimpleSpanProcessor(mockExporter);
    const span = createTestSpan("test");

    // Should not throw
    expect(() => processor.onEnd(span)).not.toThrow();

    // Error should be logged
    expect(consoleErrorSpy).toHaveBeenCalledWith(
      expect.stringContaining("SimpleSpanProcessor export failed"),
      expect.any(Error)
    );
  });

  it("should become no-op after shutdown", async () => {
    const processor = createSimpleSpanProcessor(mockExporter);

    await processor.shutdown();

    // onEnd should not call export after shutdown
    processor.onEnd(createTestSpan("test"));
    expect(mockExporter.export).not.toHaveBeenCalled();
  });
});
```
