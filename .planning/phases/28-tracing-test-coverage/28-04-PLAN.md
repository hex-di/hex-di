import { vi, describe, it, expect, beforeEach } from "vitest";
import { createJaegerExporter } from "@hex-di/tracing-jaeger";
import type { ReadableSpan } from "@opentelemetry/sdk-trace-base";

// Mock the OTel JaegerExporter
vi.mock("@opentelemetry/exporter-jaeger", () => {
  return {
    JaegerExporter: vi.fn().mockImplementation(config => {
      return {
        _config: config,
        export: vi.fn((spans, callback) => {
          // Simulate async callback
          setTimeout(() => {
            callback({ code: 0 }); // Success
          }, 0);
        }),
        shutdown: vi.fn(callback => {
          callback?.();
          return Promise.resolve();
        }),
      };
    }),
  };
});

describe("createJaegerExporter", () => {
  it("should pass configuration to underlying JaegerExporter", () => {
    const JaegerExporter = vi.mocked(require("@opentelemetry/exporter-jaeger").JaegerExporter);

    const exporter = createJaegerExporter({
      endpoint: "http://localhost:14268",
      resource: { attributes: { "service.name": "test-service" } },
    });

    expect(JaegerExporter).toHaveBeenCalledWith(
      expect.objectContaining({
        endpoint: "http://localhost:14268",
      })
    );
  });

  it("should convert callback-based export to Promise", async () => {
    const exporter = createJaegerExporter({});
    const spans: ReadableSpan[] = [createMockReadableSpan("test")];

    // Export returns a Promise
    const result = exporter.export(spans);
    expect(result).toBeInstanceOf(Promise);

    // Should resolve successfully
    await expect(result).resolves.toBeUndefined();
  });

  it("should handle export errors", async () => {
    const JaegerExporter = vi.mocked(require("@opentelemetry/exporter-jaeger").JaegerExporter);

    // Mock error response
    JaegerExporter.mockImplementationOnce(config => ({
      export: vi.fn((spans, callback) => {
        callback({ code: 1, error: new Error("Export failed") });
      }),
      shutdown: vi.fn(),
    }));

    const exporter = createJaegerExporter({});
    const spans: ReadableSpan[] = [createMockReadableSpan("test")];

    // Should reject with error
    await expect(exporter.export(spans)).rejects.toThrow("Export failed");
  });
});

function createMockReadableSpan(name: string): ReadableSpan {
  return {
    name,
    kind: 0,
    spanContext: () => ({
      traceId: "trace123",
      spanId: "span456",
      traceFlags: 1,
    }),
    startTime: [1, 0],
    endTime: [2, 0],
    status: { code: 0 },
    attributes: {},
    links: [],
    events: [],
    duration: [1, 0],
    ended: true,
    resource: { attributes: {} },
    instrumentationLibrary: { name: "test" },
    droppedAttributesCount: 0,
    droppedEventsCount: 0,
    droppedLinksCount: 0,
  } as ReadableSpan;
}

// Create minimal mock for dd-trace types
interface MockDdSpan {
  _spanContext: { _traceId: string; _spanId: string };
  _startTime?: number;
  setTag: vi.Mock;
  finish: vi.Mock;
}

interface MockDdTracer {
  startSpan: vi.Mock<MockDdSpan>;
  flush: vi.Mock;
}

// Mock dd-trace module
vi.mock("dd-trace", () => {
  return {
    default: {
      init: vi.fn(() => createMockTracer()),
    },
  };
});

function createMockTracer(): MockDdTracer {
  return {
    startSpan: vi.fn((name, options) => {
      const span: MockDdSpan = {
        _spanContext: {
          _traceId: "ddtrace123",
          _spanId: "ddspan456",
        },
        _startTime: options?.startTime,
        setTag: vi.fn(),
        finish: vi.fn(),
      };
      return span;
    }),
    flush: vi.fn(),
  };
}

describe("createDataDogBridge", () => {
  it("should create dd-trace spans with correct timing", () => {
    const bridge = createDataDogBridge({});
    const spanData = createTestSpanData("test", 1000, 2000);

    bridge.export([spanData]);

    // Verify span created with timing
    const mockTracer = require("dd-trace").default.init();
    expect(mockTracer.startSpan).toHaveBeenCalledWith(
      "test",
      expect.objectContaining({
        startTime: 1000,
      })
    );

    // Verify span finished with end time
    const mockSpan = mockTracer.startSpan.mock.results[0].value;
    expect(mockSpan.finish).toHaveBeenCalledWith(2000);
  });

  it("should convert attributes to tags", () => {
    const bridge = createDataDogBridge({});
    const spanData = createTestSpanData("test", 1000, 2000);
    spanData.attributes = {
      "http.method": "GET",
      "http.url": "/api/users",
      "custom.tag": "value",
    };

    bridge.export([spanData]);

    const mockTracer = require("dd-trace").default.init();
    const mockSpan = mockTracer.startSpan.mock.results[0].value;

    expect(mockSpan.setTag).toHaveBeenCalledWith("http.method", "GET");
    expect(mockSpan.setTag).toHaveBeenCalledWith("http.url", "/api/users");
    expect(mockSpan.setTag).toHaveBeenCalledWith("custom.tag", "value");
  });

  it("should handle parent-child relationships", () => {
    const bridge = createDataDogBridge({});

    // Export parent first
    const parentData = createTestSpanData("parent", 1000, 2000);
    bridge.export([parentData]);

    // Export child with parentSpanId
    const childData = createTestSpanData("child", 1100, 1900);
    childData.parentSpanId = parentData.context.spanId;

    bridge.export([childData]);

    const mockTracer = require("dd-trace").default.init();

    // Second call should have childOf option
    expect(mockTracer.startSpan).toHaveBeenNthCalledWith(
      2,
      "child",
      expect.objectContaining({
        childOf: expect.any(Object), // The parent span
      })
    );
  });

  it("should clean up active spans on shutdown", async () => {
    const bridge = createDataDogBridge({});

    // Export some spans
    bridge.export([createTestSpanData("span1", 1000, 2000)]);
    bridge.export([createTestSpanData("span2", 1000, 2000)]);

    // Shutdown should clear active spans
    await bridge.shutdown();

    // Further exports should not find parent spans
    const childData = createTestSpanData("child", 3000, 4000);
    childData.parentSpanId = "span1";

    bridge.export([childData]);

    const mockTracer = require("dd-trace").default.init();
    const lastCall = mockTracer.startSpan.mock.lastCall;

    // Should not have childOf since parent was cleared
    expect(lastCall[1]?.childOf).toBeUndefined();
  });
});

function createTestSpanData(name: string, startTime: number, endTime: number): SpanData {
  return {
    context: {
      traceId: "trace123",
      spanId: name,
      traceFlags: 1,
    },
    name,
    kind: "internal",
    startTime,
    endTime,
    status: "ok",
    attributes: {},
    events: [],
    links: [],
    parentSpanId: undefined,
  };
}
