<task id="1" title="Create cross-container span relationship tests">
Test parent-child span relationships when resolutions cross container boundaries.

**File:** `packages/tracing/tests/integration/instrumentation/cross-container.test.ts`

**Tests to write:**

1. **Parent-child across containers:** Root resolution triggers child container resolution
2. **Nested container chains:** A → B → C container resolution chain
3. **Sibling containers:** Multiple child containers resolved from parent
4. **Scope containers:** Scoped container resolutions maintain relationships
5. **Async boundaries:** Parent-child relationships with async resolutions

**Test setup:**

```typescript
import { createContainer, defineService, createPort } from "@hex-di/runtime";
import { createMemoryTracer } from "@hex-di/tracing/testing";
import { instrumentContainer } from "@hex-di/tracing";

// Create interconnected service graph
const ParentPort = createPort<string>("Parent");
const ChildPort = createPort<string>("Child");
const GrandchildPort = createPort<string>("Grandchild");

const parentGraph = {
  [ParentPort.key]: defineService(ParentPort).withFactory(container => {
    // This triggers child container resolution
    const child = container.resolve(ChildPort);
    return `parent-${child}`;
  }),
};

const childGraph = {
  [ChildPort.key]: defineService(ChildPort).withFactory(container => {
    // This triggers grandchild resolution
    const grandchild = container.resolve(GrandchildPort);
    return `child-${grandchild}`;
  }),
  [GrandchildPort.key]: defineService(GrandchildPort).withValue("grandchild"),
};
```

**Key assertions:**

```typescript
it("should maintain parent-child relationships across containers", () => {
  const tracer = createMemoryTracer();

  const parent = createContainer({ graph: parentGraph, name: "Parent" });
  const child = parent.createChild(childGraph, "Child");

  instrumentContainer(parent, tracer);
  instrumentContainer(child, tracer);

  parent.resolve(ParentPort);

  const spans = tracer.getCollectedSpans();

  // Find spans by name
  const parentSpan = spans.find(s => s.name === "resolve:Parent");
  const childSpan = spans.find(s => s.name === "resolve:Child");
  const grandchildSpan = spans.find(s => s.name === "resolve:Grandchild");

  // Verify parent-child relationships
  expect(childSpan?.parentSpanId).toBe(parentSpan?.context.spanId);
  expect(grandchildSpan?.parentSpanId).toBe(childSpan?.context.spanId);
});
```

<task id="2" title="Create instrumentContainerTree tests">
Test tree-wide instrumentation including dynamic child container handling.

**File:** `packages/tracing/tests/integration/instrumentation/tree-instrumentation.test.ts`

**Tests to write:**

1. **Tree walking:** All existing containers instrumented
2. **Dynamic children:** New child containers auto-instrumented
3. **Cleanup removes all hooks:** Tree-wide cleanup is complete
4. **Double-instrumentation handling:** Re-instrumenting tree is safe
5. **Inspector event subscription:** Verify child-created events handled

**Test scenarios:**

```typescript
describe("instrumentContainerTree", () => {
  let tracer: MemoryTracer;
  let rootContainer: Container;

  beforeEach(() => {
    tracer = createMemoryTracer();
    rootContainer = createContainer({ graph: rootGraph, name: "Root" });
  });

  afterEach(() => {
    void rootContainer.dispose();
  });

  it("should instrument all existing containers in tree", () => {
    // Create tree before instrumentation
    const child1 = rootContainer.createChild(child1Graph, "Child1");
    const child2 = rootContainer.createChild(child2Graph, "Child2");
    const grandchild = child1.createChild(grandchildGraph, "Grandchild");

    // Instrument entire tree
    const cleanup = instrumentContainerTree(rootContainer, rootContainer.inspector, tracer);

    // Resolve in each container
    rootContainer.resolve(RootPort);
    child1.resolve(Child1Port);
    child2.resolve(Child2Port);
    grandchild.resolve(GrandchildPort);

    // Verify spans created for all containers
    const spans = tracer.getCollectedSpans();
    expect(spans.map(s => s.name)).toContain("resolve:Root");
    expect(spans.map(s => s.name)).toContain("resolve:Child1");
    expect(spans.map(s => s.name)).toContain("resolve:Child2");
    expect(spans.map(s => s.name)).toContain("resolve:Grandchild");

    cleanup();
  });

  it("should auto-instrument dynamically created child containers", () => {
    // Instrument tree first
    const cleanup = instrumentContainerTree(rootContainer, rootContainer.inspector, tracer);

    // Create child after instrumentation
    const dynamicChild = rootContainer.createChild(childGraph, "DynamicChild");

    // Resolve in dynamic child
    dynamicChild.resolve(ChildPort);

    // Verify span created for dynamic child
    const spans = tracer.getCollectedSpans();
    expect(spans.some(s => s.name === "resolve:Child")).toBe(true);

    cleanup();
    void dynamicChild.dispose();
  });

  it("should remove all hooks when cleanup called", () => {
    const child = rootContainer.createChild(childGraph, "Child");

    const cleanup = instrumentContainerTree(rootContainer, rootContainer.inspector, tracer);

    // Verify instrumentation works
    rootContainer.resolve(RootPort);
    expect(tracer.getCollectedSpans()).toHaveLength(1);

    // Cleanup
    cleanup();
    tracer.clear();

    // Resolution should not create spans after cleanup
    rootContainer.resolve(RootPort);
    child.resolve(ChildPort);
    expect(tracer.getCollectedSpans()).toHaveLength(0);

    void child.dispose();
  });
});
```

it("should maintain span stack integrity when resolution fails", () => {
  const tracer = createMemoryTracer();

  const ErrorPort = createPort<string>("Error");
  const errorGraph = {
    [ErrorPort.key]: defineService(ErrorPort).withFactory(() => {
      throw new Error("Resolution failed");
    }),
  };

  const container = createContainer({ graph: errorGraph, name: "Test" });
  instrumentContainer(container, tracer);

  // Resolution should throw
  expect(() => container.resolve(ErrorPort)).toThrow("Resolution failed");

  // Span should be created and ended with error status
  const spans = tracer.getCollectedSpans();
  expect(spans).toHaveLength(1);

  const errorSpan = spans[0];
  expect(errorSpan.name).toBe("resolve:Error");
  expect(errorSpan.status).toBe("error");
  expect(errorSpan.events.some(e => e.name === "exception")).toBe(true);

  // Stack should be empty after error
  expect(getStackDepth()).toBe(0);

  void container.dispose();
});
