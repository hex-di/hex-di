```typescript
it("should produce same spans as instrumentContainer", () => {
  const tracer = createMemoryTracer();
  const options = { minDurationMs: 5 };

  // Test with createTracingHook
  const hooks = createTracingHook(tracer, options);
  simulateResolution(hooks.beforeResolve, hooks.afterResolve);
  const hookSpans = tracer.getCollectedSpans();

  // Test with instrumentContainer
  tracer.clear();
  const container = createMockContainer();
  instrumentContainer(container, tracer, options);
  const [, beforeResolve] = container.addHook.mock.calls[0];
  const [, afterResolve] = container.addHook.mock.calls[1];
  simulateResolution(beforeResolve, afterResolve);
  const containerSpans = tracer.getCollectedSpans();

  // Should produce identical spans
  expect(hookSpans).toEqual(containerSpans);
});
```
