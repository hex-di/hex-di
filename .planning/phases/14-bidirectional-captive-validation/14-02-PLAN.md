  <action>
    Create new file with pending constraint type definitions based on research findings:

    ```typescript
    /**
     * Pending constraint tracking for forward reference validation
     */

    import type { LifetimeLevel } from "../../../types.js";
    import type { IsNever } from "@hex-di/core";

    /**
     * Represents a pending captive constraint that needs validation
     * when the required port is eventually provided
     */
    export type PendingCaptiveConstraint<
      TDependentPort extends string,
      TDependentLevel extends LifetimeLevel,
      TRequiredPort extends string
    > = {
      readonly _tag: "PendingCaptive";
      readonly dependent: TDependentPort;
      readonly dependentLevel: TDependentLevel;
      readonly required: TRequiredPort;
    };

    /**
     * Add a new pending constraint to the union
     */
    export type AddPendingConstraint<
      TPending,
      TDependentPort extends string,
      TDependentLevel extends LifetimeLevel,
      TRequiredPort extends string
    > = TPending | PendingCaptiveConstraint<TDependentPort, TDependentLevel, TRequiredPort>;

    /**
     * Check if a newly provided port violates any pending constraints
     */
    export type CheckPendingConstraints<
      TPending,
      TNewPortName extends string,
      TNewPortLevel extends LifetimeLevel
    > = TPending extends PendingCaptiveConstraint<
      infer TDepPort,
      infer TDepLevel,
      infer TReqPort
    >
      ? TReqPort extends TNewPortName
        ? TDepLevel extends LifetimeLevel
          ? TNewPortLevel extends LifetimeLevel
            ? IsCaptiveDependency<TDepLevel, TNewPortLevel> extends true
              ? TDepPort // Captive violation found
              : never
            : never
          : never
        : never
      : never;

    /**
     * Remove satisfied constraints after port is provided
     */
    export type RemoveSatisfiedConstraints<
      TPending,
      TProvidedPort extends string
    > = TPending extends PendingCaptiveConstraint<
      infer TDepPort,
      infer TDepLevel,
      infer TReqPort
    >
      ? TReqPort extends TProvidedPort
        ? never // Remove this constraint
        : PendingCaptiveConstraint<TDepPort, TDepLevel, TReqPort> // Keep it
      : never;

    // Import IsCaptiveDependency from detection.ts
    import type { IsCaptiveDependency } from "./detection.js";
    ```

    This creates the union-based constraint storage pattern recommended in research.

  </action>
