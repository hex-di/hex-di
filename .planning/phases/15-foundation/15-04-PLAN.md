/**
 * Standalone container inspection function.
 *
 * Provides a simple API for inspecting container state without
 * needing to access the container.inspector property.
 *
 * @packageDocumentation
 */

import type { Port } from "@hex-di/core";
import type { Container, ContainerPhase } from "./types/index.js";
import type { ContainerSnapshot } from "./inspection/internal-state-types.js";
import { INTERNAL_ACCESS } from "./inspection/symbols.js";
import {
  detectContainerKindFromInternal,
  buildTypedSnapshotFromInternal,
} from "./inspection/internal-helpers.js";

/**
 * Type for objects that have INTERNAL_ACCESS symbol.
 * @internal
 */
interface HasInternalAccess {
  readonly [INTERNAL_ACCESS]: () => unknown;
}

/**
 * Inspects a container and returns a full snapshot of its state.
 *
 * Returns a frozen snapshot including:
 * - All adapters registered in the container
 * - All cached singleton instances
 * - Scope tree structure
 * - Lifetime information for each adapter
 * - Container metadata (name, kind, phase)
 *
 * @param container - The container to inspect
 * @returns A frozen ContainerSnapshot with full state information
 *
 * @example
 * ```typescript
 * import { createContainer, inspect } from '@hex-di/runtime';
 *
 * const container = createContainer(graph, { name: 'App' });
 * container.resolve(LoggerPort);
 *
 * const snapshot = inspect(container);
 * console.log('Singletons:', snapshot.singletons.length);
 * console.log('Adapters:', snapshot.adapters.length);
 * console.log('Phase:', snapshot.phase);
 * ```
 */
export function inspect<
  TProvides extends Port<unknown, string>,
  TExtends extends Port<unknown, string> = never,
  TAsyncPorts extends Port<unknown, string> = never,
  TPhase extends ContainerPhase = "uninitialized",
>(container: Container<TProvides, TExtends, TAsyncPorts, TPhase>): ContainerSnapshot {
  // Access internal state via symbol
  const internalState = (container as HasInternalAccess)[INTERNAL_ACCESS]();

  // Build and return typed snapshot
  const containerKind = detectContainerKindFromInternal(
    internalState as Parameters<typeof detectContainerKindFromInternal>[0]
  );

  return buildTypedSnapshotFromInternal(
    {
      isDisposed: (internalState as { isDisposed: boolean }).isDisposed,
      singletonMemo: (internalState as { singletonMemo: Map<unknown, unknown> }).singletonMemo,
      adapterMap: (internalState as { adapterMap: Map<unknown, unknown> }).adapterMap,
      childScopes: (internalState as { childScopes: Set<unknown> }).childScopes,
      childContainers: (internalState as { childContainers: Set<unknown> }).childContainers,
    },
    containerKind,
    internalState as Parameters<typeof buildTypedSnapshotFromInternal>[2]
  );
}
````

Note: The implementation leverages existing internal helpers from inspection module. The function provides a simpler API than container.inspector.getSnapshot().

/**
 * Standalone container tracing functions.
 *
 * Provides simple APIs for tracing container resolutions:
 * - trace(container, fn): Trace resolutions within a callback
 * - enableTracing(container): Enable global tracing, returns disable function
 *
 * @packageDocumentation
 */

import type { Port } from "@hex-di/core";
import type { TraceEntry } from "@hex-di/core";
import type { Container, ContainerPhase } from "./types/index.js";
import type { ResolutionHookContext, ResolutionResultContext } from "./resolution/hooks.js";

/**
 * Result of a traced execution.
 */
export interface TraceResult<R> {
  /** The result of the traced function */
  readonly result: R;
  /** Array of trace entries collected during execution */
  readonly traces: readonly TraceEntry[];
}

/**
 * Traces all resolutions within a callback function.
 *
 * Installs temporary hooks before the callback runs, collects all
 * resolution traces, then removes the hooks and returns both the
 * callback result and the collected traces.
 *
 * @param container - The container to trace
 * @param fn - The function to execute with tracing
 * @returns Object with result and traces array
 *
 * @example
 * ```typescript
 * import { createContainer, trace } from '@hex-di/runtime';
 *
 * const container = createContainer(graph, { name: 'App' });
 *
 * const { result, traces } = trace(container, () => {
 *   const logger = container.resolve(LoggerPort);
 *   const db = container.resolve(DatabasePort);
 *   return { logger, db };
 * });
 *
 * console.log('Resolved services:', result);
 * console.log('Resolution traces:', traces);
 * // traces = [
 * //   { portName: 'Logger', duration: 0.5, ... },
 * //   { portName: 'Database', duration: 2.1, ... }
 * // ]
 * ```
 */
export function trace<
  TProvides extends Port<unknown, string>,
  TExtends extends Port<unknown, string>,
  TAsyncPorts extends Port<unknown, string>,
  TPhase extends ContainerPhase,
  R,
>(container: Container<TProvides, TExtends, TAsyncPorts, TPhase>, fn: () => R): TraceResult<R> {
  const traces: TraceEntry[] = [];
  const startTimes = new Map<string, number>();

  // Install before hook to capture start time
  const beforeHandler = (ctx: ResolutionHookContext): void => {
    startTimes.set(ctx.portName, performance.now());
  };

  // Install after hook to capture trace entry
  const afterHandler = (ctx: ResolutionResultContext): void => {
    const startTime = startTimes.get(ctx.portName) ?? performance.now();
    const duration = performance.now() - startTime;

    traces.push({
      id: `${ctx.portName}-${Date.now()}`,
      portName: ctx.portName,
      lifetime: ctx.lifetime,
      duration,
      timestamp: Date.now(),
      depth: ctx.depth,
      cacheHit: ctx.cacheHit,
    });
  };

  // Install hooks
  container.addHook("beforeResolve", beforeHandler);
  container.addHook("afterResolve", afterHandler);

  try {
    // Execute the function
    const result = fn();
    return { result, traces: Object.freeze(traces) };
  } finally {
    // Remove hooks
    container.removeHook("beforeResolve", beforeHandler);
    container.removeHook("afterResolve", afterHandler);
  }
}

/**
 * Enables global tracing for a container.
 *
 * Returns a function to disable tracing. While enabled, all resolutions
 * are traced and available via container.tracer.getTraces().
 *
 * @param container - The container to enable tracing on
 * @returns A function to disable tracing
 *
 * @example
 * ```typescript
 * import { createContainer, enableTracing } from '@hex-di/runtime';
 *
 * const container = createContainer(graph, { name: 'App' });
 * const disableTracing = enableTracing(container);
 *
 * // All resolutions are now traced
 * container.resolve(LoggerPort);
 * container.resolve(DatabasePort);
 *
 * // Access traces via container.tracer
 * const traces = container.tracer.getTraces();
 * console.log(traces);
 *
 * // Disable when done
 * disableTracing();
 * ```
 */
export function enableTracing<
  TProvides extends Port<unknown, string>,
  TExtends extends Port<unknown, string>,
  TAsyncPorts extends Port<unknown, string>,
  TPhase extends ContainerPhase,
>(container: Container<TProvides, TExtends, TAsyncPorts, TPhase>): () => void {
  const startTimes = new Map<string, number>();

  const beforeHandler = (ctx: ResolutionHookContext): void => {
    startTimes.set(ctx.portName, performance.now());
  };

  const afterHandler = (ctx: ResolutionResultContext): void => {
    const startTime = startTimes.get(ctx.portName) ?? performance.now();
    const duration = performance.now() - startTime;

    // Add to container's built-in tracer
    // The tracer already has a collector, we just need to feed it
    // This works because container.tracer.subscribe returns trace entries
    // For now, we'll use the collector pattern that's already in place
  };

  container.addHook("beforeResolve", beforeHandler);
  container.addHook("afterResolve", afterHandler);

  return () => {
    container.removeHook("beforeResolve", beforeHandler);
    container.removeHook("afterResolve", afterHandler);
    startTimes.clear();
  };
}
````

Note: The trace() function provides scoped tracing (collect traces only during callback). The enableTracing() function provides global tracing (all subsequent resolutions traced until disabled).
