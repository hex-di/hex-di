    ```typescript
    /**
     * Override context for temporary service substitution.
     * @internal
     */
    import type { Port, Adapter } from "@hex-di/core";
    import type { BaseContainer } from "./base-impl.js";

    export class OverrideContext {
      private readonly parent: BaseContainer;
      private readonly overrides: Map<Port<unknown, string>, Adapter<unknown>>;
      private readonly memoMap: Map<Port<unknown, string>, unknown>;

      constructor(
        parent: BaseContainer,
        overrides: Record<Port<unknown, string>, Adapter<unknown>>
      ) {
        this.parent = parent;
        this.overrides = new Map(Object.entries(overrides).map(
          ([port, adapter]) => [port as Port<unknown, string>, adapter]
        ));
        this.memoMap = new Map();
      }

      /**
       * Resolves a port within the override context.
       * Checks overrides first, then delegates to parent if not overridden.
       */
      resolve<T>(port: Port<T, string>): T {
        // Check if this port is overridden
        const overrideAdapter = this.overrides.get(port);
        if (overrideAdapter) {
          // Check memo for cached instance
          if (this.memoMap.has(port)) {
            return this.memoMap.get(port) as T;
          }

          // Create instance using override adapter
          const instance = this.createInstance(overrideAdapter, port);

          // Cache if appropriate (based on lifetime)
          if (this.shouldCache(overrideAdapter)) {
            this.memoMap.set(port, instance);
          }

          return instance as T;
        }

        // Not overridden, delegate to parent
        return this.parent.resolve(port);
      }

      private createInstance(
        adapter: Adapter<unknown>,
        port: Port<unknown, string>
      ): unknown {
        // Get dependencies
        const deps = adapter.requires || [];
        const resolvedDeps = deps.map(dep => this.resolve(dep));

        // Call factory with resolved dependencies
        return adapter.factory(...resolvedDeps);
      }

      private shouldCache(adapter: Adapter<unknown>): boolean {
        // Cache singleton and scoped services, not transient
        return adapter.lifetime !== 'transient';
      }

      /**
       * Executes a function within this override context.
       */
      execute<R>(fn: () => R): R {
        // Store current context if needed for nested resolution
        return fn();
      }
    }
    ```

    This implementation creates an isolated context with its own memoization map, ensuring override instances don't pollute the parent container's cache.
