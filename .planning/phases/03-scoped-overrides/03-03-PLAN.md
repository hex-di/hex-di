     * Request scope for HTTP request isolation.

    import type { Port, InferService } from "@hex-di/core";
    import type { Scope } from "../types.js";
    import { ScopeImpl } from "./scope-impl.js";
    import type { BaseContainer } from "../container/base-impl.js";

    export class RequestScope<
      TProvides extends Port<unknown, string>,
      TAsyncPorts extends Port<unknown, string>,
      TPhase extends "uninitialized" | "initialized"
    > extends ScopeImpl<TProvides, TAsyncPorts, TPhase> {
      private readonly requestId: string;
      private readonly requestMemo: Map<Port<unknown, string>, unknown>;

      constructor(
        container: BaseContainer,
        name?: string
      ) {
        super(container, name || `request-${Date.now()}`);
        this.requestId = crypto.randomUUID ? crypto.randomUUID() : `req-${Date.now()}-${Math.random()}`;
        this.requestMemo = new Map();
      }

      /**
       * Resolves a service within the request scope.
       * Request-scoped services are cached per request.
       */
      resolve<P extends TProvides>(port: P): InferService<P> {
        // Check if this is a request-scoped service
        const adapter = this.getAdapter(port);
        if (adapter?.lifetime === 'request') {
          // Check request-specific memo
          if (this.requestMemo.has(port)) {
            return this.requestMemo.get(port) as InferService<P>;
          }

          // Create new instance for this request
          const instance = super.resolve(port);
          this.requestMemo.set(port, instance);
          return instance;
        }

        // Delegate to normal scope resolution for non-request services
        return super.resolve(port);
      }

      /**
       * Gets the unique request ID for this scope.
       */
      getRequestId(): string {
        return this.requestId;
      }

      /**
       * Disposes the request scope and clears request-specific instances.
       */
      async dispose(): Promise<void> {
        // Clear request-specific memo
        this.requestMemo.clear();

        // Call parent dispose
        await super.dispose();
      }
